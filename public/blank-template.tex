\begin{document}


\thispagestyle{empty}
\noindent
\begin{tikzpicture}[remember picture, overlay, shift={(current page.north west)}]
  \pgfmathsetmacro{\ml}{0}
  \pgfmathsetmacro{\pw}{\paperwidth/1in - 2*\ml}
  \pgfmathsetmacro{\ph}{\paperheight/1in - 2*\ml}
  \pgfmathsetmacro{\cw}{\pw/4}
  \pgfmathsetmacro{\ch}{\ph/4}
  \pgfmathsetmacro{\pbw}{\cw * 0.75}
  \definecolor{gridgray}{gray}{0.9}

  % Define your 16 cell texts here
  \def\celltext#1{%
    \ifcase#1
     {}\or{}\or{}\or{}\or
     {}\or{}\or{}\or{}\or
     {}\or{}\or{}\or{}\or
     {}\or{}\or{}\or{}\or
\fi}

  % Place text nodes (no borders)
  \foreach \r in {0,...,3} {
    \foreach \c in {0,...,3} {
      \pgfmathtruncatemacro{\idx}{\r*4+\c}
      \node[minimum width=\cw in, minimum height=\ch in,
            anchor=north west, inner sep=4pt, align=center,
            text width=\cw in - 8pt]
        at (\ml in + \c*\cw in, -\ml in - \r*\ch in)
        {\parbox[c]{\pbw in}{\raggedright\celltext{\idx}}};
    }
  }

  % Horizontal lines (dotted) — 3 interior lines
  \foreach \r in {1,2,3} {
    \draw[gridgray, densely dotted, thick]
      (\ml in, -\ml in - \r*\ch in)
      -- (\ml in + \pw in, -\ml in - \r*\ch in);
  }

  % Vertical lines (solid) — 3 interior lines
  % Columns 1 and 3 are solid, column 2 (middle) is dotted
  \foreach \c in {1,3} {
    \draw[gridgray, thick]
      (\ml in + \c*\cw in, -\ml in)
      -- (\ml in + \c*\cw in, -\ml in - \ph in);
  }
  % Middle vertical line (dotted)
  \draw[gridgray, densely dotted, thick]
    (\ml in + 2*\cw in, -\ml in)
    -- (\ml in + 2*\cw in, -\ml in - \ph in);

\end{tikzpicture}
